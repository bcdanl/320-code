2.2 Q2 & Q3

set.seed(1234)

df <- df |> 
  mutate(rnd = runif(n()),
         brand = factor(brand,
                        levels = 
                          c("dominicks", "minute.maid", "tropicana")),
         ad_status = factor(ad_status,
                            levels = c("No Ad", "Ad")))

dtrain <- df |> 
  filter(rnd > .3) |> 
  select(-rnd)

dtest <- df |> 
  filter(rnd <= .3) |> 
  select(-rnd)

         data = dtrain)

m3 <- lm(log(sales) ~ brand * ad_status * log(price),
         data = dtrain)
